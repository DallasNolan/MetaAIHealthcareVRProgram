The Gemini API proxy — keeps your API key off the Quest headset

A
Create folder and install packages
Run in PowerShell

cd C:\mr-hud-server
npm init -y

B
Create .env file
Get your API key from aistudio.google.com → Get API Key (free tier)

Type exactly one line:

GEMINI_API_KEY=your-key-here
!Never share or commit this file. The key gives access to your Gemini quota.
C
Create server.js
notepad C:\mr-hud-server\server.js
require('dotenv').config();
const express = require('express');
const cors = require('cors');
const { GoogleGenerativeAI } = require('@google/generative-ai');

const app = express();
app.use(cors());
app.use(express.json());

const genai = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);
const model = genai.getGenerativeModel({ model: 'gemini-2.5-flash-lite' });

app.post('/query', async (req, res) => {
  try {
    const result = await model.generateContent(req.body.message);
    res.json({ reply: result.response.text() });
  } catch (err) {
    res.status(500).json({ reply: 'Error: ' + err.message });
  }
});

app.listen(3000, () => console.log('Gemini proxy running on :3000'));
✓Test it: run node server.js then in another PowerShell: Invoke-WebRequest -Uri http://localhost:3000/query -Method POST -Headers @{"Content-Type"="application/json"} -Body '{"message":"say hello"}'
D
Create ask_gemini.ps1
PowerShell chat interface — use this instead of VR controller input
notepad C:\mr-hud-server\ask_gemini.ps1
while ($true) {
    $query = Read-Host "Ask Gemini"
    if ($query -eq "exit") { break }
    $body = '{"message":"' + $query + '"}'
    $response = Invoke-WebRequest -Uri http://localhost:3000/query `
        -Method POST `
        -Headers @{"Content-Type"="application/json"} `
        -Body $body
    $json = $response.Content | ConvertFrom-Json
    Write-Host "Gemini: $($json.reply)" -ForegroundColor Green
    Write-Host ""
}

Create ekg_sender.py
Reads heart rate from Excel Data Streamer, sends to Quest via UDP
notepad C:\mr-hud-server\ekg_sender.py
import socket, json, time, xlwings as xw

QUEST_IP     = "192.168.137.123"   # Quest IP on laptop hotspot — update if changed
QUEST_PORT   = 5001
MONITOR_PORT = 5003
BROADCAST    = "192.168.137.255"

udp = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
udp.setsockopt(socket.SOL_SOCKET, socket.SO_BROADCAST, 1)
udp.bind(("192.168.137.1", 0))    # must have hotspot ON before running

print("Connecting to Excel...")
try:
    wb = xw.books.active
    ws = wb.sheets.active
    print(f"Connected to: {wb.name}")
except:
    print("Could not connect to Excel — open Excel with Data Streamer running first")
    exit()

print("Streaming heart rate...")

bpm_history = []
heart_rate = 0

while True:
    try:
        raw = ws.range('D5').value    # Vernier Hand-Grip outputs BPS in CH3 = column D row 5
        if raw is not None:
            bpm = float(raw) * 60    # convert beats per second → beats per minute
            if 50 < bpm < 150:       # filter noise
                bpm_history.append(bpm)
                if len(bpm_history) > 10:
                    bpm_history.pop(0)
                heart_rate = int(sum(bpm_history) / len(bpm_history))

            payload = json.dumps({
                "ekg":  round(float(raw), 4),
                "hr":   heart_rate,
                "seq":  int(time.time())
            }).encode()

            udp.sendto(payload, (QUEST_IP, QUEST_PORT))
            udp.sendto(payload, (BROADCAST, MONITOR_PORT))
            print(f"HR: {heart_rate} bpm  |  Raw: {raw:.4f} BPS")
        else:
            print("No data in D5 — is Data Streamer collecting?")

    except Exception as e:
        print(f"Error: {e}")

    time.sleep(0.1)

Create laptop_detector.py
YOLOv8 person detection via webcam — sends bounding boxes to Quest
notepad C:\mr-hud-server\laptop_detector.py
import cv2, socket, json, time
from ultralytics import YOLO

QUEST_IP   = "192.168.137.123"
QUEST_PORT = 5002

model = YOLO("yolov8n.pt")   # downloads ~6MB model automatically on first run
sock  = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
cap   = cv2.VideoCapture(1)  # change to 0 if camera index 1 does not work

print("Person detector running — press Q in the window to quit")

while True:
    ret, frame = cap.read()
    if not ret:
        print("No camera frame — retrying...")
        time.sleep(1)
        continue

    results = model(frame, classes=[0], verbose=False, conf=0.7)
    boxes = []

    for r in results[0].boxes:
        x1, y1, x2, y2 = r.xyxy[0].tolist()
        boxes.append({"x": x1, "y": y1, "w": x2 - x1, "h": y2 - y1})

    if boxes:
        sock.sendto(json.dumps({"boxes": boxes}).encode(), (QUEST_IP, QUEST_PORT))
        print(f"Detected {len(boxes)} person(s)")

    cv2.imshow("Person Detection", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
!If the camera window shows the wrong camera, change cv2.VideoCapture(1) to cv2.VideoCapture(0).
